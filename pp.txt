#include <stdio.h>
#define INF 1e9

int main() {
    int n, i;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], rt[n], priority[n], ct[n], tat[n], wt[n];
    int completed = 0, time = 0;
    float avg_tat = 0, avg_wt = 0;

    // Input Arrival Time, Burst Time, and Priority
    for (i = 0; i < n; i++) {
        printf("Enter Arrival Time for Process P%d: ", i + 1);
        scanf("%d", &at[i]);
        printf("Enter Burst Time for Process P%d: ", i + 1);
        scanf("%d", &bt[i]);
        rt[i] = bt[i]; // remaining time
        printf("Enter Priority for Process P%d (lower number = higher priority): ", i + 1);
        scanf("%d", &priority[i]);
    }

    while (completed < n) {
        int idx = -1;
        int highest_priority = INF;

        // Find highest-priority process among arrived
        for (i = 0; i < n; i++) {
            if (at[i] <= time && rt[i] > 0 && priority[i] < highest_priority) {
                highest_priority = priority[i];
                idx = i;
            }
        }

        if (idx != -1) {
            rt[idx]--; // execute for 1 unit
            time++;

            if (rt[idx] == 0) {
                ct[idx] = time;
                tat[idx] = ct[idx] - at[idx];
                wt[idx] = tat[idx] - bt[idx];
                avg_tat += tat[idx];
                avg_wt += wt[idx];
                completed++;
            }
        } else {
            time++; // CPU idle
        }
    }

    // Display results
    printf("\nProcess\tAT\tBT\tPriority\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t\t%d\t%d\t%d\n", i + 1, at[i], bt[i], priority[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f\n", avg_tat / n);
    printf("Average Waiting Time: %.2f\n", avg_wt / n);

    return 0;
}
