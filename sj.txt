#include <stdio.h>

int main() {
    int n, i, j;
    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], ct[n], tat[n], wt[n], temp_bt[n];
    int completed[n];
    float avg_tat = 0, avg_wt = 0;

    // Input Arrival Time and Burst Time
    for (i = 0; i < n; i++) {
        printf("Enter Arrival Time for Process P%d: ", i + 1);
        scanf("%d", &at[i]);
        printf("Enter Burst Time for Process P%d: ", i + 1);
        scanf("%d", &bt[i]);
        temp_bt[i] = bt[i];  // store original burst time
        completed[i] = 0;    // process not completed yet
    }

    int time = 0, completed_count = 0;

    while (completed_count < n) {
        int idx = -1;
        int min_bt = 1e9;

        // Find process with shortest burst time among arrived processes
        for (i = 0; i < n; i++) {
            if (at[i] <= time && !completed[i] && bt[i] < min_bt) {
                min_bt = bt[i];
                idx = i;
            }
        }

        if (idx != -1) {
            time += bt[idx];
            ct[idx] = time;
            tat[idx] = ct[idx] - at[idx];
            wt[idx] = tat[idx] - temp_bt[idx];
            avg_tat += tat[idx];
            avg_wt += wt[idx];
            completed[idx] = 1;
            completed_count++;
        } else {
            time++; // if no process has arrived, increment time
        }
    }

    // Display results
    printf("\nProcess\tAT\tBT\tCT\tTAT\tWT\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t%d\t%d\t%d\n", i + 1, at[i], temp_bt[i], ct[i], tat[i], wt[i]);
    }

    printf("\nAverage Turnaround Time: %.2f\n", avg_tat / n);
    printf("Average Waiting Time: %.2f\n", avg_wt / n);

    return 0;
}
